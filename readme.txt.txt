mkdir - створити теку;
cd "назва директорії"- для зміни поточного місця розташування робочого каталогу;
ls - показує файли в директорії;
ls -la - те саме що ls приховані файли в директорі;
git init - ініцілізуємо директорію;
git status - показує зміни в директорії;
git add назва файлу та розширення - додає файл в директорію;
git commit -m "повідомлення з описом" Записує зміни в файлі в історію версій, створюючи новий коміт
git diff - назва файлу- що змінено;
git log - послідовність комітів;
git branch - гілки;
git branch нова назва гілки  - нова назва гілки;
git checkout назва гілки або коміту - переміщає на певну гілку або коміт;
git clone https://логін:токінг@силка та репозиторій - клонує репозиторій;
git remote - список віддалених репозиторіїв;
git push назва зі списків - відправляє на github;
git clone посилання - клонування; 
git push --set-upstream origin назва гілки - оновлення в репозиторії;
